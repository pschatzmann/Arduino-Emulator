193 uint16_t blockingRead(void* data, int len, int timeout = 1000) {
194 int offset = 0;
195 long start = millis();
196 while (offset < len && (millis() - start) < timeout) {
197 int n = io->readBytes((char*)data + offset, len - offset);
198 offset += n;
199 }
200 return offset;
201 }
