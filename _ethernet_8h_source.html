326 // resolves the address and returns sockaddr_in
327 IPAddress resolveAddress(const char* address, uint16_t port) {
328 struct sockaddr_in serv_addr4;
329 memset(&serv_addr4, 0, sizeof(serv_addr4));
330 serv_addr4.sin_family = AF_INET;
331 serv_addr4.sin_port = htons(port);
332 if (::inet_pton(AF_INET, address, &serv_addr4.sin_addr) <= 0) {
333 // Not an IP, try to resolve hostname
334 struct hostent* he = ::gethostbyname(address);
335 if (he == nullptr || he->h_addr_list[0] == nullptr) {
336 Logger.error(WIFICLIENT, "Hostname resolution failed");
337 serv_addr4.sin_addr.s_addr = 0;
338 } else {
339 memcpy(&serv_addr4.sin_addr, he->h_addr_list[0], he->h_length);
340 }
341 }
342 return IPAddress(serv_addr4.sin_addr.s_addr);
343 }
