107 bool begin_(int port = 0) {
108 if (port > 0) _port = port;
109 _status = wl_status_t::WL_DISCONNECTED;
110
111 // create server socket
112 if ((server_fd = socket(AF_INET, SOCK_STREAM, 0)) < 0) {
113 // error("socket failed");
114 _status = wl_status_t::WL_CONNECT_FAILED;
115 return false;
116 }
117
118 // Reuse address after restart
119 int iSetOption = 1;
120 setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, (char*)&iSetOption,
121 sizeof(iSetOption));
122
123 // Set SO_REUSEPORT for better port reuse
124 setsockopt(server_fd, SOL_SOCKET, SO_REUSEPORT, (char*)&iSetOption,
125 sizeof(iSetOption));
126
127 // config socket
128 server_addr.sin_family = AF_INET;
129 server_addr.sin_addr.s_addr = INADDR_ANY;
130 server_addr.sin_port = htons(_port);
131
132 // bind socket to port
133 while (::bind(server_fd, (struct sockaddr*)&server_addr,
134 sizeof(server_addr)) < 0) {
135 // error("bind failed");
136 //_status = wl_status_t::WL_CONNECT_FAILED;
137 Logger.error("bind failed");
138 // return false;
139 delay(1000);
140 }
141
142 // listen for connections
143 if (::listen(server_fd, 10) < 0) {
144 // error("listen failed");
145 _status = wl_status_t::WL_CONNECT_FAILED;
146 Logger.error("listen failed");
147 return false;
148 }
149
150 // Add to active servers list for signal handling
151 active_servers().push_back(this);
152 _status = wl_status_t::WL_CONNECTED;
