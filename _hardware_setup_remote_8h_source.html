70 bool begin(Stream* s, bool asDefault = true, bool doHandShake = true) {
71 p_stream = s;
72
73 i2c.setStream(s);
74 spi.setStream(s);
75 gpio.setStream(s);
76
77 // setup global objects
78 if (asDefault) {
79 SPI.setSPI(&spi);
80 Wire.setI2C(&i2c);
81 GPIO.setGPIO(&gpio);

83
84 if (doHandShake) {

86 }

88 }

91 void begin(int port, bool asDefault) {
92 this->port = port;

97 void begin(bool asDefault = true) {
98 is_default_objects_active = asDefault;
99 if (p_stream == nullptr) {
100 default_stream.begin(port);
101 handShake(&default_stream);
102 IPAddress ip = default_stream.remoteIP();
103 int remote_port = default_stream.remotePort();
104 default_stream.setTarget(ip, remote_port);
105 default_stream.write((const uint8_t*)"OK", 2);
106 default_stream.flush();
107 begin(&default_stream, asDefault, false);
108 } else {
109 begin(p_stream, asDefault, true);
110 }
111 }
112
113 void end() {
114 if (is_default_objects_active) {
115 GPIO.setGPIO(nullptr);
116 SPI.setSPI(nullptr);
117 Wire.setI2C(nullptr);
118 }
119 if (p_stream == &default_stream) {
120 default_stream.stop();
121 }
122 }

137 void handShake(Stream* s) {
138 while (true) {
139 Logger.warning("HardwareSetup", "waiting for device...");
140 try {
141 // we wait for the Arduino to send us the Arduino-Emulator string
142 if (s->available() >= 16) {
143 char buffer[30];
144 int len = s->readBytes(buffer, 18);
145 buffer[len] = 0;
146 if (strncmp(buffer, "Arduino-Emulator", 16)) {
147 Logger.info("WiFiUDPStream", "device found!");
148 break;
149 } else {
150 Logger.info("WiFiUDPStream", "unknown command", buffer);
151 }
152 }
153 delay(10000);
154 } catch (const std::exception& ex) {
155 Logger.error("WiFiUDPStream", ex.what());
156 }
157 }
158 }
