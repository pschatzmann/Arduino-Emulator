54class Servo {
55 public:
56 Servo()
57 : _pin(SERVO_NO_PIN),
58 _min(SERVO_DEFAULT_MIN_US),
59 _max(SERVO_DEFAULT_MAX_US),
60 _pulseUs((SERVO_DEFAULT_MIN_US + SERVO_DEFAULT_MAX_US) / 2) {}
61
62 ~Servo() {
63 if (attached()) detach();
64 }

71 uint8_t attach(int pin) {
72 return attach(pin, SERVO_DEFAULT_MIN_US, SERVO_DEFAULT_MAX_US);

82 uint8_t attach(int pin, int min, int max) {
83 _pin = pin;
84 _min = std::max(min, 500); // pigpio lower bound
85 _max = std::min(max, 2500); // pigpio upper bound
86 gpioSetMode(_pin, PI_OUTPUT);
87 gpioServo(_pin, _pulseUs);
88 return 0;
89 }
90
94 void detach() {
95 if (attached()) {
96 gpioServo(_pin, 0); // 0 = stop pulses
97 _pin = SERVO_NO_PIN;
98 }
99 }
100
105 void write(int value) {
106 if (value <= 200) {
107 value = std::max(0, std::min(value, 180));
108 writeMicroseconds(angleToPulse(value));
109 } else {
110 writeMicroseconds(value);
111 }
112 }
113
118 void writeMicroseconds(int us) {
119 _pulseUs = std::max(_min, std::min(us, _max));
120 if (attached()) gpioServo(_pin, _pulseUs);
121 }
122
127 int read() const {
128 return pulseToAngle(_pulseUs);
129 }
130
135 int readMicroseconds() const {
136 return _pulseUs;
137 }
138
143 bool attached() const {
144 return _pin != SERVO_NO_PIN;
145 }
146
147 private:
148 int _pin;
149 int _min;
150 int _max;
151 int _pulseUs;
152
154 int angleToPulse(int angle) const {
155 return _min + (angle * (_max - _min)) / 180;
156 }
157
159 int pulseToAngle(int us) const {
160 if (_max == _min) return 0;
161 int angle = ((us - _min) * 180) / (_max - _min);
162 return std::max(0, std::min(angle, 180));
163 }
164};
