91 uint8_t attach(int pin, int min, int max) {
92 _pin = pin;
93 _min = min;
94 _max = max;
95 _gpio.pinMode(_pin, OUTPUT);

167 int _pulseToAngle(int us) const {
168 if (_max == _min) return 0;
169 int angle = ((us - _min) * 180) / (_max - _min);
170 return std::max(0, std::min(angle, 180));
171 }

181 void _writePulse(int us) {
182 int pwm_channel = -1;
183 if (_pin == 18 || _pin == 12) pwm_channel = 0;
184 else if (_pin == 19 || _pin == 13) pwm_channel = 1;
185 if (pwm_channel < 0) return;
186
187 // Export the PWM channel (no-op if already exported)
188 FILE* fexp = fopen("/sys/class/pwm/pwmchip0/export", "w");
189 if (fexp) { fprintf(fexp, "%d", pwm_channel); fclose(fexp); }
190
191 char period_path[64], duty_path[64], enable_path[64];
192 snprintf(period_path, sizeof(period_path),
193 "/sys/class/pwm/pwmchip0/pwm%d/period", pwm_channel);
194 snprintf(duty_path, sizeof(duty_path),
195 "/sys/class/pwm/pwmchip0/pwm%d/duty_cycle", pwm_channel);
196 snprintf(enable_path, sizeof(enable_path),
197 "/sys/class/pwm/pwmchip0/pwm%d/enable", pwm_channel);
198
199 // Set period to 20,000,000 ns (50 Hz)
200 FILE* fp = fopen(period_path, "w");
201 if (fp) { fprintf(fp, "%lu", SERVO_PERIOD_NS); fclose(fp); }
202
203 // Convert us to ns and write duty cycle (0 ns when disabling)
204 unsigned long duty_ns = (us > 0) ? (unsigned long)us * 1000UL : 0UL;
205 FILE* fd = fopen(duty_path, "w");
206 if (fd) { fprintf(fd, "%lu", duty_ns); fclose(fd); }
207
208 // Enable or disable PWM output
209 FILE* fe = fopen(enable_path, "w");
210 if (fe) { fprintf(fe, "%d", us > 0 ? 1 : 0); fclose(fe); }
211 }
