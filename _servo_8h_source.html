93 uint8_t attach(int pin, int min, int max) {
94 _pin = pin;
95 _min = min;
96 _max = max;
97 _gpio.pinMode(_pin, OUTPUT);
98 _gpio.analogWriteFrequency(_pin, 50); // Fix to 50 Hz for servo control
99 _writePulse(_pulseUs);
100 return 0;
101 }
102
106 void detach() {
107 if (attached()) {
108 _writePulse(0);
109 _pin = SERVO_NO_PIN;
110 }
111 }

117 void write(int value) {
118 if (value <= 200) {
119 value = std::max(0, std::min(value, 180));
120 writeMicroseconds(_angleToPulse(value));
121 } else {
122 writeMicroseconds(value);
123 }
124 }

132 if (attached()) _writePulse(_pulseUs);
133 }

162
164 int _angleToPulse(int angle) const {
165 return _min + (angle * (_max - _min)) / 180;
166 }
167
169 int _pulseToAngle(int us) const {
170 if (_max == _min) return 0;
171 int angle = ((us - _min) * 180) / (_max - _min);
172 return std::max(0, std::min(angle, 180));
173 }
174
183 void _writePulse(int us) {
184 int pwm_channel = -1;
185 if (_pin == 18 || _pin == 12) pwm_channel = 0;
186 else if (_pin == 19 || _pin == 13) pwm_channel = 1;
187 if (pwm_channel < 0) return;
188
189 // Export the PWM channel (no-op if already exported)
190 FILE* fexp = fopen("/sys/class/pwm/pwmchip0/export", "w");
191 if (fexp) { fprintf(fexp, "%d", pwm_channel); fclose(fexp); }
192
193 char period_path[64], duty_path[64], enable_path[64];
194 snprintf(period_path, sizeof(period_path),
195 "/sys/class/pwm/pwmchip0/pwm%d/period", pwm_channel);
196 snprintf(duty_path, sizeof(duty_path),
197 "/sys/class/pwm/pwmchip0/pwm%d/duty_cycle", pwm_channel);
198 snprintf(enable_path, sizeof(enable_path),
199 "/sys/class/pwm/pwmchip0/pwm%d/enable", pwm_channel);
200
201 // Set period to 20,000,000 ns (50 Hz)
202 FILE* fp = fopen(period_path, "w");
203 if (fp) { fprintf(fp, "%lu", SERVO_PERIOD_NS); fclose(fp); }
204
205 // Convert us to ns and write duty cycle (0 ns when disabling)
206 unsigned long duty_ns = (us > 0) ? (unsigned long)us * 1000UL : 0UL;
207 FILE* fd = fopen(duty_path, "w");
208 if (fd) { fprintf(fd, "%lu", duty_ns); fclose(fd); }
209
210 // Enable or disable PWM output
211 FILE* fe = fopen(enable_path, "w");
212 if (fe) { fprintf(fe, "%d", us > 0 ? 1 : 0); fclose(fe); }
213 }
