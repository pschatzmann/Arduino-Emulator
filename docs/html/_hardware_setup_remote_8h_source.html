97  void begin(bool asDefault = true) {

104  default_stream.setTarget(ip, remote_port);
105  default_stream.write((const uint8_t*)"OK", 2);

137  void handShake(Stream* s) {
138  while (true) {
139  Logger.warning("HardwareSetup", "waiting for device...");
140  try {
141  // we wait for the Arduino to send us the Arduino-Emulator string
142  if (s->available() >= 16) {
143  char buffer[30];
144  int len = s->readBytes(buffer, 18);
145  buffer[len] = 0;
146  if (strncmp(buffer, "Arduino-Emulator", 16)) {
147  Logger.info("WiFiUDPStream", "device found!");
148  break;
149  } else {
150  Logger.info("WiFiUDPStream", "unknown command", buffer);
151  }
152  }
153  delay(10000);
154  } catch (const std::exception& ex) {
155  Logger.error("WiFiUDPStream", ex.what());
156  }
157  }
158  }
